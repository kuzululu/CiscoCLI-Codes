Cisco Command

!@BASIC SECURITY
enable secret pass
line console 0
password cisco
login
exit
line vty 0 15
password cisco
login
exit
service password-encryption

!@Create VLAN 
conf t
vlan #
name VlanName

!@assign port
int vlan #
int range port# range / int port#
switchport mode access
switchport access vlan #
!veriy vlans
sh vlan br

!remove specific port in vlan
conf t
interface fa0/9
no switchport access vlan #

!or 
conf t
interface fa0/9
switchport access vlan 1

!@trunking if many switch 
!notest: if error occurred “Command rejected: An interface whose trunk encapsulation is "Auto" can not be configured to "trunk" mode” type this cli first “switchport trunk encapsulation dot1q” before “switchport mode trunk”

int port# 
switchport mode trunk
!@verify
Sh cdp neigh !to show the trunking links
sh int trunk

!@CLI for SVI Switch 

!@SVI Configuration or creating a gateway
conf t
int vlan #
ip address (g.w ip or first-host subnet mask)
no shut

!@verify SVI
sh ip int br | ex una
sh ip int vlan #

!@optional enable Inter-vlan Routing if want to access each vlan ID
 ip routing
!@verify 
 sh ip route

@STILL IN SVI SWITCH
!@DHCP EXCLUDED ADDRESS
 Ip dhcp excluded-address (ip address to be excluded in each vlans)

!@DHCP POOL 
 ip dhcp pool VLANNAME-POOL
network (ip address subnet mask)
default-router (first host or gateway ip)
dns-server (first host or gateway ip or consise dns 8.8.8.8)
domain-name ServerNAme

!optional if neccessary 
!for end devices like servers, pc, printer
conf t
int port#
ip add dhcp
end

!show ip of end devices
sh ip int br | ex una

!@Verify DCHP POOL
sh ip dhcp binding
sh ip dhcp pool
show running-config | include excluded-address
show run | section dhcp
!remove dhcp pool VLANname
no ip dhcp pool vlanName

!@EXTENDED ACL
 ip  access-list extended Name-Policy
deny ip SOURCE_NETWORK SOURCE_WILDCARD DEST_NETWORK DEST_WILDCARD
permit ip any any

!@APPLY ACL
 int vlan #
 ip access-group Name-Policy in

!@Importante: Kung naka-apply pa ang ACL sa interface, magandang tanggalin muna:
interface vlan #
no ip access-group NAme-Policy in

@Pagkatapos saka:
no ip access-list extended HR-Policy

!@remove acl policy
conf t
no ip access-list extended Name-policy

!or remove specific entry
conf t
ip access-list extended HR-Policy
no deny ip 172.16.100.32 0.0.0.15 172.16.100.0 0.0.0.15

!@VERIFY ACL
 sh access-lists
 sh ip access-lists
 sh running-config | inc access-group
 show running-config
show running-config interface vlan # !specific vlan
show ip interface vlan # !makikita rito ang inbound(in)
